111 // class template Conversion
112 // Figures out the conversion relationships between two types
113 // Invocations (T and U are types):
114 // a) Conversion<T, U>::exists
115 // returns (at compile time) true if there is an implicit conversion from T
116 // to U (example: Derived to Base)
117 // b) Conversion<T, U>::exists2Way
118 // returns (at compile time) true if there are both conversions from T
119 // to U and from U to T (example: int to char and back)
120 // c) Conversion<T, U>::sameType
121 // returns (at compile time) true if T and U represent the same type
122 //
123 // Caveat: might not work if T and U are in a private inheritance hierarchy.
125 
126  template <class T, class U>
127  struct Conversion
128  {
129  typedef Private::ConversionHelper<T, U> H;
130 #ifndef __MWERKS__
131  enum { exists = sizeof(typename H::Small) == sizeof((H::Test(H::MakeT()))) };
132 #else
133  enum { exists = false };
134 #endif
135  enum { exists2Way = exists && Conversion<U, T>::exists };
136  enum { sameType = false };
137  };
138 
139  template <class T>
140  struct Conversion<T, T>
141  {
142  enum { exists = 1, exists2Way = 1, sameType = 1 };
143  };
